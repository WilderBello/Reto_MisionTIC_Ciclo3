# Marriott_Hotel

Comandos:

requerimientos: 
  - pip freeze > requirements.txt ### Obtener requerimientos.
  - pip install -r requirements.txt ### Instalar los requerimientos.

Windows:
  - .\env\Scripts\activate
  - set FLASK_APP=app ### no es necesario
  - set FLASK_DEBUG=1
  - flask run
  - .\env\Scripts\deactivate
